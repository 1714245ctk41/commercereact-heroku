function checkemail($value){
	$pattern='#^[a-z][a-z0-9_\.]{4,31}@[a-z0-9]{2,}(\.[a-z0-9]){1,2}#';
	$flag=false;
	if(preg_match($pattern, $value)==true)
		{$flag=true;}
	return $flag;
}
function checkname($value){
	$pattern='#^[a-z0-9_][a-z0-9_\.\s]{4,31}#';

		$flag=true;

}
